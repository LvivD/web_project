          TAG_NAME {
            rule;

          }

          .CLASS_NAME {

          #ID {

Also it's combinations are possible:


          TAG_NAME.CLASS_NAME {
            rule;
            rule;
          }

          TAG_NAME.CLASS_NAME TAG_NAME {
            rule;
            rule;
          }
